// dashboard.js import { auth } from "./firebase-init.js"; import { onAuthStateChanged, signOut } from "https://www.gstatic.com/firebasejs/9.0.0/firebase-auth.js"; onAuthStateChanged(auth, (user) => { if (user) { // Pengguna terautentikasi console.log("Pengguna login:", user.email); } else { // Pengguna tidak terautentikasi window.location.href = "login.html"; } }); // Tombol logout const logoutBtn = document.getElementById("logout"); logoutBtn.addEventListener("click", () => { signOut(auth) .then(() => { window.location.href = "login.html"; }) .catch((error) => { console.error("Error saat logout:", error); }); });
